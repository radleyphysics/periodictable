\newcommand{\CommonElementTextFormat}[4]
{
% 1 - Z; 2 - A; 3 - SymbolX; 4 - Name
  \begin{minipage}{2.5cm}
    \centering
      {{#2}} \rule{0pt}{1em}\\%
      \vspace{0.5em}
      {\textbf{#3}}\\%
      \vspace{0.3em}
      {{#4}}\\
      \vspace{0.4em}
      {\textbf{#1}}%
  \end{minipage}
}
