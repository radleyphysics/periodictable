\newcommand{\CommonElementTextFormat}[4]
{
% 1 - Z; 2 - A; 3 - SymbolX; 4 - Name
  \begin{minipage}{2.2cm}
    \centering
      {{#2}} \rule{0pt}{1em}\\%
      \vspace{0.5em}
      {\textbf{#3}}\\%
      \vspace{0.3em}
      {{#4}}\\
      \vspace{0.2em}
      {\textbf{#1}}%
  \end{minipage}
}

%    \centering
%      {\textbf{#1} \hfill #2}%
%      \linebreak \linebreak
%      {\textbf{#3}}%
%      \linebreak \linebreak
%      {{#4}}

\newcommand{\NaturalElementTextFormat}[4]
{
  \CommonElementTextFormat{#1}{#2}{\Huge {#3}}{#4}
}

\newcommand{\SyntheticElementTextFormat}[4]
{
  \CommonElementTextFormat{#1}{#2}{\Huge #3}{#4}
}
